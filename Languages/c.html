C Cheatsheet By DevOasis
  1. Boilerplate Code: 
    #include<stdio.h> //header files
int main() //main function
{
    // Your code here
    return(0); //returning value to int main()
}
  2. printf function: 
    printf("Hello World!");
  3. Compile hello.c file with gcc: 
    $ gcc hello.c -o hello
  4. Run the compiled binary hello: 
    $ ./hello
  5. Variables: 
    int myNum = 15;

int myNum2; // do not assign, then assign
myNum2 = 15;

int myNum3 = 15; // myNum3 is 15
myNum3 = 10;     // myNum3 is now 10

float myFloat = 5.99; // floating point number
char myLetter = 'D';  // character

int x = 5;
int y = 6;
int sum = x + y; // add variables to sum

// declare multiple variables
int x = 5, y = 6, z = 50;
  6. Constants: 
    const int minutesPerHour = 60;
const float PI = 3.14;
  7. Comments: 
    // this is a comment
printf("Hello World!"); // Can comment anywhere in file

/*Multi-line comment, print Hello World!
to the screen, it's awesome */
  8. Strings: 
    char greetings[] = "Hello World!";
printf("%s", greetings);
  9. Conditions: 
    int time = 20;
if (time < 18) {
    printf("Goodbye!");
} else {
    printf("Good evening!");
}
// Output -> "Good evening!"
int time = 22;
if (time < 10) {
    printf("Good morning!");
} else if (time < 20) {
    printf("Goodbye!");
} else {
    printf("Good evening!");
}
// Output -> "Good evening!"
  10. Switch: 
    int day = 4;

switch (day) {
    case 3: printf("Wednesday"); break;
    case 4: printf("Thursday"); break;
    default:
    printf("Weekend!");
}
// output -> "Thursday" (day 4)
  11. While loop: 
    int i = 0;

while (i < 5) {
    printf("%d\n", i);
    i++;
}
  12. Do-While loop: 
    int i = 0;

do {
printf("%d\n", i);
i++;
} while (i < 5);
  13. For loop: 
    for (int i = 0; i < 5; i++) {
printf("%d\n", i);
