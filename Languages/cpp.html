  7. Else if statement: 
    int score = 99;
if (score == 100) {
    std::cout << "Superb";
}
else if (score >= 90) {
    std::cout << "Excellent";
}
else if (score >= 80) {
    std::cout << "Very Good";
}
else if (score >= 70) {
    std::cout << "Good";
}
else if (score >= 60)
    std::cout << "OK";
else
    std::cout << "What?";
  8. Ternary Operator: 
               ┌── True ──┐
Result = Condition ? Exp1 : Exp2;
            └───── False ─────┘
//Example
int x = 3, y = 5, max;
max = (x > y) ? x : y;

// Outputs: 5
std::cout << max << std::endl;
  9. Switch Statement: 
    int num = 2;
switch (num) {
    case 0:
        std::cout << "Zero";
        break;
    case 1:
        std::cout << "One";
        break;
    case 2:
        std::cout << "Two";
        break;
    case 3:
        std::cout << "Three";
        break;
    default:
        std::cout << "What?";
        break;
}
  10. While loop: 
    int i = 0;
while (i < 6) {
    std::cout << i++;
}

// Outputs: 012345
  11. Do-While loop: 
    int i = 1;
do {
    std::cout << i++;
} while (i <= 5);

// Outputs: 12345
