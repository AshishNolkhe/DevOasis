  console.log('Big');
} else if (size > 20) {
  console.log('Medium');
} else if (size > 4) {
  console.log('Small');

  console.log('Tiny');

  10. Loops & closures: 
    // Prints 3 thrice, not what we meant.

    setTimeout(_ => console.log(i), 10);
}

// Prints 0, 1 and 2, as expected.

  setTimeout(_ => console.log(j), 10);
}
